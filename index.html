form.addEventListener('submit', function(e) { e.preventDefault(); const churchName = document.getElementById('churchName').value; const contactPerson = document.getElementById('contactPerson').value; const appDate = document.getElementById('appDate').value; const phoneNumber = document.getElementById('phoneNumber').value; const notes = document.getElementById('notes').value; // Save locally const appointment = { id: Date.now(), churchName, contactPerson, date: new Date(appDate).toLocaleString(), rawDate: appDate, phoneNumber, notes }; let appointments = JSON.parse(localStorage.getItem('churchApps')) || []; appointments.push(appointment); localStorage.setItem('churchApps', JSON.stringify(appointments)); // Create Google Calendar event URL const startDate = new Date(appDate); // Default 1-hour appointment const endDate = new Date(startDate.getTime() + 60 * 60 * 1000); function formatGoogleDate(date) { return date.toISOString().replace(/-|:|\.\d+/g, ''); } const eventTitle = encodeURIComponent(`Church Meeting - ${churchName}`); const eventDetails = encodeURIComponent( `Contact: ${contactPerson} Phone: ${phoneNumber} Notes: ${notes}` ); const calendarUrl = `https://calendar.google.com/calendar/render?action=TEMPLATE` + `&text=${eventTitle}` + `&dates=${formatGoogleDate(startDate)}/${formatGoogleDate(endDate)}` + `&details=${eventDetails}`; // Open Google Calendar event creation page window.open(calendarUrl, '_blank'); form.reset(); displayAppointments(); });
